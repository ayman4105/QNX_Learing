
project(HelloWorld_Client)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# QNX specific
set(CMAKE_C_COMPILER qcc)
set(CMAKE_CXX_COMPILER q++)
set(CMAKE_C_COMPILER_TARGET gcc_ntoaarch64le)
set(CMAKE_CXX_COMPILER_TARGET gcc_ntoaarch64le_cxx)

# Prevent finding host packages
set(CMAKE_FIND_ROOT_PATH "$ENV{QNX_TARGET};$ENV{QNX_STAGE}")
set(CMAKE_FIND_ROOT_PATH_MODE_PROGRAM NEVER)
set(CMAKE_FIND_ROOT_PATH_MODE_LIBRARY ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_INCLUDE ONLY)
set(CMAKE_FIND_ROOT_PATH_MODE_PACKAGE ONLY)

# Find packages (QNX target only)
find_package(CommonAPI REQUIRED)
find_package(CommonAPI-SomeIP REQUIRED)
find_package(vsomeip3 REQUIRED)

message(STATUS "CommonAPI found: ${CommonAPI_FOUND}")
message(STATUS "CommonAPI-SomeIP found: ${CommonAPI-SomeIP_FOUND}")
message(STATUS "vsomeip3 found: ${vsomeip3_FOUND}")

include_directories(
    ${CMAKE_CURRENT_SOURCE_DIR}/SRC
    ${CMAKE_CURRENT_SOURCE_DIR}/fidl/src-gen
)

set(GENERATED_SOMEIP_SOURCES
    ${CMAKE_CURRENT_SOURCE_DIR}/fidl/src-gen/v1/commonapi/HelloWorldSomeIPDeployment.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/fidl/src-gen/v1/commonapi/HelloWorldSomeIPProxy.cpp
    ${CMAKE_CURRENT_SOURCE_DIR}/fidl/src-gen/v1/commonapi/HelloWorldSomeIPStubAdapter.cpp
)

# Client only (QNX RPi5)
add_executable(HelloWorldClient
    SRC/Client.cpp
    ${GENERATED_SOMEIP_SOURCES}
)

target_link_libraries(HelloWorldClient
    CommonAPI
    CommonAPI-SomeIP
    vsomeip3
)
